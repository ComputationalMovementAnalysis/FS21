euclid <- function(x1,y1,x2,y2){ return(sqrt((x1-x2)2+(y1-y2)2)) } turning_angle <- function(x,y,lead_lag = 1){ if(length(x) < 3){stop(“Minimum length of x and y is 3”)} if(length(x) != length(y)){stop(“x and y must be of the same length”)} p1x <- lag(x,lead_lag) p1y <- lag(y,lead_lag) p2x <- x p2y <- y p3x <- lead(x,lead_lag) p3y <- lead(y,lead_lag) p12 <- euclid(p1x,p1y,p2x,p2y) p13 <- euclid(p1x,p1y,p3x,p3y) p23 <- euclid(p2x,p2y,p3x,p3y) rad <- acos((p122+p232-p13^2)/(2p12p23)) grad <- (rad180)/pi grad[p12 == 0 | p23 == 0] <- NA d <- (p3x-p1x)(p2y-p1y)-(p3y-p1y)(p2x-p1x) d <- ifelse(d == 0,1,d) d[d>0] <- 1 d[d<0] <- -1 d[d==0] <- 1 turning <- gradd*-1+180 return(turning) }
